36 void OdbcWrapper::handleError(SQLHANDLE handle, SQLSMALLINT handleType, SQLRETURN retCode) {
37 OdbcLogger::logInfo("Entering handleError");
38 SQLWCHAR sqlState[6], msg[SQL_MAX_MESSAGE_LENGTH];
39 SQLINTEGER nativeError;
40 SQLSMALLINT msgLen;
41 bool errorLogged = false;
42
43 while (m_odbc->SQLGetDiagRec(handleType, handle, 1, sqlState, &nativeError,
44 msg, sizeof(msg) / sizeof(SQLWCHAR), &msgLen) == SQL_SUCCESS) {
45 errorLogged = true;
46 // Convert SQLWCHAR* to std::wstring for logging
47 std::wstring sqlStateStr(reinterpret_cast<wchar_t*>(sqlState));
48 std::wstring msgStr(reinterpret_cast<wchar_t*>(msg));
49
50 // Convert std::wstring to std::string for spdlog (with error handling)
51 std::string sqlStateNarrow;
52 std::string msgNarrow;
53 try {
54 std::wstring_convert<std::codecvt_utf8<wchar_t>> converter;
55 sqlStateNarrow = converter.to_bytes(sqlStateStr);
56 msgNarrow = converter.to_bytes(msgStr);
57 } catch (const std::exception&) {
58 // Fallback to simple conversion if UTF-8 conversion fails
59 sqlStateNarrow = std::string(sqlStateStr.begin(), sqlStateStr.end());
60 msgNarrow = std::string(msgStr.begin(), msgStr.end());
61 }
62
63 OdbcLogger::logError(fmt::format("SQLSTATE: {}, Message: {}, Native Error: {}",
64 sqlStateNarrow, msgNarrow, nativeError));
65
66 if (retCode == SQL_ERROR) {
67 throw std::runtime_error("ODBC Error: " + msgNarrow);
68 }
69 }
70
71 // If SQL_ERROR but couldn't get diagnostic info, still throw
72 if (retCode == SQL_ERROR && !errorLogged) {
73 OdbcLogger::logError("ODBC Error occurred but diagnostic information unavailable");
74 throw std::runtime_error("ODBC Error: Unable to retrieve diagnostic information");
75 }

162 OdbcLogger::logInfo("Entering fetchResults");

169 SQLSMALLINT numCols;

171
172 while (m_odbc->SQLFetch(m_hStmt) == SQL_SUCCESS) {
173 std::vector<std::wstring> row;
174 for (SQLSMALLINT i = 1; i <= numCols; i++) {
175 SQLWCHAR buffer[1024];
176 SQLLEN indicator;
177 m_odbc->SQLGetData(m_hStmt, i, SQL_C_WCHAR, buffer, sizeof(buffer), &indicator);
178
179 if (indicator != SQL_NULL_DATA) {
180 row.push_back(std::wstring(reinterpret_cast<wchar_t*>(buffer)));
181 } else {
182 row.push_back(L"NULL");
183 }
184 }
185 results.push_back(row);
186 }
